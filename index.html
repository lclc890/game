Program 2: String Operations (string.c)

#include <stdio.h>
#include <string.h>

void stlen(char[]);
void stcmp(char[],char[]);
void stcat(char[],char[]);

void main()
{
char st1[20],st2[20];
printf("Enter first string:\n");
scanf("%s",st1);
printf("Enter second string:\n");
scanf("%s",st2);
stlen(st1);
stlen(st2);
stcmp(st1,st2);
stcat(st1,st2);
}

void stlen(char s[])
{
printf("Length of %s is %d\n",s,strlen(s));
}

void stcmp(char a[],char b[])
{
int r=strcmp(a,b);
if(r==0)
printf("Strings are equal\n");
else if(r>0)
printf("%s is greater\n",a);
else
printf("%s is greater\n",b);
}

void stcat(char a[],char b[])
{
strcat(a,b);
printf("Concatenated string: %s\n",a);
}

Program 3: Structure – Book Search

#include <stdio.h>
#include <string.h>

struct book {
int book_id;
char title[50];
char author[50];
};

int main()
{
int n,i,found=0;
char search[50];
printf("Enter number of books:");
scanf("%d",&n);
struct book b[n];

for(i=0;i<n;i++)
{
printf("ID:");
scanf("%d",&b[i].book_id);
printf("Title:");
scanf(" %[^\n]",b[i].title);
printf("Author:");
scanf(" %[^\n]",b[i].author);
}

printf("Enter title to search:");
scanf(" %[^\n]",search);

for(i=0;i<n;i++)
{
if(strcmp(b[i].title,search)==0)
{
printf("Book Found\n");
found=1;
}
}
if(!found)
printf("Book not found\n");
return 0;
}

Program 5: GCD and LCM

#include <stdio.h>
int gcd(int a,int b)
{
while(b!=0)
{
int t=b;
b=a%b;
a=t;
}
return a;
}

int main()
{
int a,b;
scanf("%d%d",&a,&b);
printf("GCD=%d\n",gcd(a,b));
printf("LCM=%d\n",(a*b)/gcd(a,b));
return 0;
}

Program 7: Floyd’s Triangle

#include <stdio.h>
int main()
{
int r,n=1;
scanf("%d",&r);
for(int i=1;i<=r;i++)
{
for(int j=1;j<=i;j++)
printf("%d ",n++);
printf("\n");
}
return 0;
}
